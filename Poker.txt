function parseRange(rangeStr) {
    if (!rangeStr) return [];
    let hands = new Set();
    let parts = rangeStr.split(',').map(s => s.trim());
    parts.forEach(p => {
        if (p.length === 3 && p[0] === p[1] && p.endsWith('+')) {
            let start = ranks.indexOf(p[0]);
            for (let i = start; i < ranks.length; i++) hands.add(ranks[i] + ranks[i]);
        } else if (p.length === 4 && p.endsWith('+')) {
            let top = ranks.indexOf(p[0]);
            let start = ranks.indexOf(p[1]);
            for (let i = start; i <= top; i++) {
                if (ranks[i] !== ranks[top]) hands.add(ranks[top] + ranks[i] + p[2]);
            }
        } else { hands.add(p); }
    });
    return Array.from(hands);
}

// Extract hero position from any key format
function heroFromKey(key) {
    if (key.startsWith("RFI_"))      return key.slice(4);
    if (key.startsWith("RFI6_"))     return key.slice(5);
    if (key.startsWith("FACING_"))   return key.slice(7).split("_VS_")[0];
    if (key.startsWith("FACING6_"))  return key.slice(8).split("_VS_")[0];
    if (key.startsWith("3BET6_"))    return key.slice(6).split("_VS_")[0];
    if (key.startsWith("VS_3BET_"))  return key.slice(8).split("_VS_")[0];
    if (key.startsWith("VS3BET6_"))  return key.slice(9).split("_VS_")[0];
    return null;
}

// Extract villain position from any key format
function villainFromKey(key) {
    if (key.startsWith("RFI_") || key.startsWith("RFI6_")) return null;
    const vsIdx = key.indexOf("_VS_");
    if (vsIdx === -1) return null;
    return key.slice(vsIdx + 4);
}

function dealHand() {
    answered = false;

    // Pick a random position
    const heroPos = allPositions[Math.floor(Math.random() * allPositions.length)];

    // Find valid scenarios for that position
    const validKeys = Object.keys(pokerRanges).filter(key => heroFromKey(key) === heroPos);

    if (validKeys.length === 0) { dealHand(); return; }

    activeScenarioKey = validKeys[Math.floor(Math.random() * validKeys.length)];
    const scenarioData = pokerRanges[activeScenarioKey];

    // Deal two cards
    const suits = ['♠','♥','♦','♣'];
    let r1 = ranks[Math.floor(Math.random() * ranks.length)];
    let r2 = ranks[Math.floor(Math.random() * ranks.length)];
    let s1 = suits[Math.floor(Math.random() * suits.length)];
    let s2 = suits[Math.floor(Math.random() * suits.length)];
    while (r1 === r2 && s1 === s2) { s2 = suits[Math.floor(Math.random() * suits.length)]; }

    let i1 = ranks.indexOf(r1), i2 = ranks.indexOf(r2);
    let hi = i1 >= i2 ? r1 : r2, lo = i1 >= i2 ? r2 : r1;
    currentHand = hi === lo ? hi + lo : hi + lo + (s1 === s2 ? 's' : 'o');

    // Determine correct action
    correctAction = "Fold";
    if (scenarioData.type === "RFI") {
        if (parseRange(scenarioData.raise).includes(currentHand)) correctAction = "Raise";
    } else {
        let aggrKey = scenarioData.type === "VS_3BET" ? "4-Bet" : "3-Bet";
        if (parseRange(scenarioData[aggrKey]).includes(currentHand)) correctAction = aggrKey;
        else if (parseRange(scenarioData["Call"]).includes(currentHand)) correctAction = "Call";
    }

    // Update visuals
    updateTable(heroPos, activeScenarioKey, r1, s1, r2, s2);
    updateButtons(scenarioData.type);

    // Update position label
    const posNames = { UTG:'UTG', UTG1:'UTG+1', UTG2:'UTG+2', LJ:'LJ', HJ:'HJ', CO:'CO', BTN:'BTN', SB:'SB', BB:'BB', MP:'MP' };
    document.getElementById('pos-label').textContent = posNames[heroPos] || heroPos;

    // Update hand cards in action panel
    const sc1 = (s1==='♥'||s1==='♦') ? 'red' : 'black';
    const sc2 = (s2==='♥'||s2==='♦') ? 'red' : 'black';
    const hc1 = document.getElementById('hc1');
    const hc2 = document.getElementById('hc2');
    hc1.className = `hand-card-big ${sc1}`; hc1.textContent = r1+s1; hc1.style.animationName='none'; void hc1.offsetWidth; hc1.style.animationName='';
    hc2.className = `hand-card-big ${sc2}`; hc2.textContent = r2+s2; hc2.style.animationName='none'; void hc2.offsetWidth; hc2.style.animationName='';
    document.getElementById('hand-notation').textContent = currentHand;

    document.getElementById('feedback').textContent = '';
    document.getElementById('feedback').className = 'feedback';

    document.getElementById('action-buttons').style.display = 'flex';
    document.getElementById('next-btn').style.display = 'none';
}

        centerBadgeText = 'Facing RFI';
        centerPosText = oppPos || '?';
        centerVerbText = 'open-raised voor jou (2.5 BB)';
    } else if (isVs3bet) {
        badgeText = '↑↑ 3-Bet · ~9bb';
        badgeClass = 'badge-3bet';
        centerBadgeText = 'Facing 3-Bet';
        centerPosText = oppPos || '?';
        centerVerbText = '3-bet jouw open-raise (~9 BB)';
    } else {
        centerBadgeText = 'RFI';
        centerPosText = heroPos;
        centerVerbText = 'iedereen foldt naar jou';
    }

    // Update center scenario card
    const card = document.getElementById('scenario-card');
    const badge = document.getElementById('scenario-badge');
    const posEl = document.getElementById('scenario-pos');
    const verbEl = document.getElementById('scenario-verb');
    card.className = `scenario-card ${scenarioType === 'rfi' ? 'rfi-mode' : scenarioType === 'facing' ? 'facing-mode' : 'vs3bet-mode'}`;
    badge.textContent = centerBadgeText;
    posEl.textContent = centerPosText;
    verbEl.textContent = centerVerbText;

    // Hero seat (MP maps visually to UTG2)
    const heroSeat = document.getElementById('seat-' + seatId(heroPos));
    if (heroSeat) {
        heroSeat.classList.add('hero');
        const rc1 = (s1==='♥'||s1==='♦') ? 'red' : 'black';
        const rc2 = (s2==='♥'||s2==='♦') ? 'red' : 'black';
        heroSeat.querySelector('.seat-cards').innerHTML =
            `<div class="card ${rc1}" style="animation-delay:0s">${r1}${s1}</div>
             <div class="card ${rc2}" style="animation-delay:0.08s">${r2}${s2}</div>`;
    }

    // Villain seat + badge + chips (MP maps visually to UTG2)
    if (oppPos) {
        const oppSeat = document.getElementById('seat-' + seatId(oppPos));
        if (oppSeat) {
            oppSeat.classList.add('villain');
            oppSeat.querySelector('.seat-cards').innerHTML =
                `<div class="card card-back" style="animation-delay:0s"></div>
                 <div class="card card-back" style="animation-delay:0.08s"></div>`;
            const vbadge = document.getElementById('vbadge-' + seatId(oppPos));
            if (vbadge) { vbadge.textContent = badgeText; vbadge.className = `villain-action-badge ${badgeClass}`; }
        }

        // Show chip stacks (MP maps to UTG2)
        showChips(seatId(oppPos), isVs3bet ? '3bet' : 'raise');
        if (isVs3bet) {
            showChips(seatId(heroPos), 'raise');
        }
    }
}

// ── 3D CHIP TOWER RENDERER ───────────────────────────────────────────────────
const CHIP_DEF = {
    white: { top:'#ddd',    mid:'#bbb',    side:'#999',   dark:'#777',  stripe:'#fff',  shine:'rgba(255,255,255,0.7)' },
    red:   { top:'#e03030', mid:'#b01818', side:'#880808',dark:'#550000',stripe:'#ff8888',shine:'rgba(255,120,120,0.55)' },
    blue:  { top:'#2060d8', mid:'#1040a0', side:'#082878',dark:'#041850',stripe:'#80b0ff',shine:'rgba(100,160,255,0.55)' },
    green: { top:'#22aa44', mid:'#147828', side:'#085018',dark:'#042810',stripe:'#66ee99',shine:'rgba(80,220,130,0.5)' },
    black: { top:'#383838', mid:'#222',    side:'#111',   dark:'#000',  stripe:'#666',  shine:'rgba(150,150,150,0.35)' },
    gold:  { top:'#d4a020', mid:'#a07010', side:'#6a4800',dark:'#3a2800',stripe:'#ffe080',shine:'rgba(255,230,100,0.6)' },
};

// Draw a single 3D chip as SVG elements.
// cx, cy = center of TOP face. W=chip diameter, SH=visible side height, EY=top ellipse y-radius
function chipSVG(cx, cy, W, SH, EY, colorKey) {
    const c = CHIP_DEF[colorKey] || CHIP_DEF.white;
    const rx = W / 2;
    const id = `c${colorKey}${Math.random().toString(36).slice(2,6)}`;

    // Stripe dots positions (on side, 4 dots at 90° intervals visible from front)
    const dotR = W * 0.07;
    const dotY = cy + EY * 0.2;  // slightly below center of top face

    return `
      <defs>
        <linearGradient id="sg${id}" x1="0%" y1="0%" x2="100%" y2="0%">
          <stop offset="0%"   stop-color="${c.dark}"/>
          <stop offset="18%"  stop-color="${c.side}"/>
          <stop offset="50%"  stop-color="${c.top}" stop-opacity="0.85"/>
          <stop offset="80%"  stop-color="${c.mid}"/>
          <stop offset="100%" stop-color="${c.dark}"/>
        </linearGradient>
        <radialGradient id="tg${id}" cx="38%" cy="35%" r="62%">
          <stop offset="0%"   stop-color="${c.shine.replace('rgba','rgba').replace(/,[^,)]+\)$/,',1)')}"/>
          <stop offset="45%"  stop-color="${c.top}"/>
          <stop offset="100%" stop-color="${c.dark}"/>
        </radialGradient>
        <radialGradient id="sh${id}" cx="50%" cy="100%" r="60%">
          <stop offset="0%"   stop-color="rgba(0,0,0,0.45)"/>
          <stop offset="100%" stop-color="rgba(0,0,0,0)"/>
        </radialGradient>
      </defs>
      <!-- ground shadow -->
      <ellipse cx="${cx}" cy="${cy + SH + EY * 0.55}" rx="${rx * 0.9}" ry="${EY * 0.3}"
               fill="rgba(0,0,0,0.3)"/>
      <!-- side cylinder -->
      <rect x="${cx - rx}" y="${cy}" width="${W}" height="${SH + EY * 0.6}" rx="${rx * 0.08}"
            fill="url(#sg${id})"/>
      <!-- bottom cap ellipse -->
      <ellipse cx="${cx}" cy="${cy + SH}" rx="${rx}" ry="${EY * 0.6}"
               fill="${c.side}"/>
      <!-- stripe dots on side (left / right) -->
      <circle cx="${cx - rx + dotR * 1.8}" cy="${dotY + SH * 0.4}" r="${dotR}" fill="${c.stripe}" opacity="0.9"/>
      <circle cx="${cx + rx - dotR * 1.8}" cy="${dotY + SH * 0.4}" r="${dotR}" fill="${c.stripe}" opacity="0.9"/>
      <circle cx="${cx - rx * 0.3}" cy="${dotY + SH * 0.38}" r="${dotR * 0.85}" fill="${c.stripe}" opacity="0.6"/>
      <circle cx="${cx + rx * 0.3}" cy="${dotY + SH * 0.38}" r="${dotR * 0.85}" fill="${c.stripe}" opacity="0.6"/>
      <!-- top face -->
      <ellipse cx="${cx}" cy="${cy}" rx="${rx}" ry="${EY}"
               fill="url(#tg${id})" stroke="${c.dark}" stroke-width="0.5"/>
      <!-- inset ring on top face -->
      <ellipse cx="${cx}" cy="${cy}" rx="${rx * 0.72}" ry="${EY * 0.72}"
               fill="none" stroke="${c.stripe}" stroke-width="1.4" opacity="0.45"/>
      <!-- shine highlight on top -->
      <ellipse cx="${cx - rx * 0.22}" cy="${cy - EY * 0.28}" rx="${rx * 0.32}" ry="${EY * 0.25}"
               fill="${c.shine}" opacity="0.75"/>
    `;
}

// Build a tower of N chips stacked on top of each other, rendered as one SVG.
function drawChipTower(colorStack) {
    const CW = 38;       // chip diameter px
    const SH = 7;        // visible side height per chip px
    const EY = 7;        // top ellipse y-radius px
    const N = colorStack.length;
    const PAD = 6;

    // Total SVG height: top ellipse + N*SH + bottom shadow
    const svgW = CW + PAD * 2;
    const svgH = EY + N * SH + EY * 0.9 + PAD * 2;
    const cx = svgW / 2;

    let parts = [];
    // Draw from bottom to top so upper chips overlay lower
    for (let i = N - 1; i >= 0; i--) {
        // cy = center of top face for chip i
        const cy = PAD + EY + i * SH;
        parts.push(chipSVG(cx, cy, CW, SH, EY, colorStack[i]));
    }

    return `<svg xmlns="http://www.w3.org/2000/svg"
                 width="${svgW}" height="${svgH}"
                 class="chip-svg"
                 style="overflow:visible">${parts.join('')}</svg>`;
}

function showChips(pos, type) {
    const stack = document.getElementById('chips-' + pos);
    if (!stack) return;
    stack.innerHTML = '';

    let colors, labelText, labelClass;
    if (type === 'raise') {
        colors = ['blue', 'white', 'white'];      // 3 chips = 2.5 BB raise
        labelText = '2.5 BB';
        labelClass = 'raise-label';
    } else {
        colors = ['black', 'red', 'red', 'blue', 'blue'];  // 5 chips = ~9 BB 3-bet
        labelText = '~9 BB';
        labelClass = 'bet3-label';
    }

    stack.innerHTML =
        drawChipTower(colors) +
        `<div class="chip-label ${labelClass}">${labelText}</div>`;

    // Trigger animation
    stack.classList.remove('visible');
    requestAnimationFrame(() => stack.classList.add('visible'));
}

function updateButtons(type) {
    document.getElementById('btn-raise').style.display = type === "RFI"     ? 'block' : 'none';
    document.getElementById('btn-3bet').style.display  = type === "FACING"  ? 'block' : 'none';
    document.getElementById('btn-4bet').style.display  = type === "VS_3BET" ? 'block' : 'none';
    document.getElementById('btn-call').style.display  = type !== "RFI"     ? 'block' : 'none';
    document.getElementById('btn-fold').style.display  = 'block';
    document.getElementById('next-btn').style.display  = 'none';
}

function checkAction(playerAction) {
    if (answered) return;
    answered = true;
    total++;

    const feedback = document.getElementById('feedback');
    const heroSeat = document.querySelector('.seat.hero');

    if (playerAction === correctAction) {
        feedback.innerHTML = `✓ Correct! <em>${currentHand}</em> is een <strong>${correctAction}</strong>.`;
        feedback.className = 'feedback correct';
        score++;
        if (heroSeat) heroSeat.classList.add('result-correct');
    } else {
        feedback.innerHTML = `✗ Fout. Met <em>${currentHand}</em> had je moeten <strong>${correctAction}</strong>.`;
        feedback.className = 'feedback incorrect';
        if (heroSeat) heroSeat.classList.add('result-wrong');
    }

    document.getElementById('score').textContent = score;
    document.getElementById('total').textContent = total;

    // Accuracy
    const pct = Math.round((score / total) * 100);
    document.getElementById('accuracy').textContent = pct + '%';
    document.getElementById('acc-fill').style.width = pct + '%';

    document.getElementById('action-buttons').style.display = 'flex';
    // Hide action buttons, show next
    ['btn-raise','btn-3bet','btn-4bet','btn-call','btn-fold'].forEach(id => {
        document.getElementById(id).style.display = 'none';
    });
    document.getElementById('next-btn').style.display = 'block';
}

// Keyboard shortcuts
document.addEventListener('keydown', e => {
    if (answered) {
        if (e.code === 'Space' || e.code === 'Enter') { e.preventDefault(); dealHand(); }
        return;
    }
    const type = pokerRanges[activeScenarioKey]?.type;
    if (e.key === 'r' || e.key === 'R') { if (type==='RFI') checkAction('Raise'); }
    if (e.key === '3') { if (type==='FACING') checkAction('3-Bet'); }
    if (e.key === '4') { if (type==='VS_3BET') checkAction('4-Bet'); }
    if (e.key === 'c' || e.key === 'C') { if (type!=='RFI') checkAction('Call'); }
    if (e.key === 'f' || e.key === 'F') checkAction('Fold');
});
